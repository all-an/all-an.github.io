Union / intersection type

A union type says a value may be one of several alternatives, while an intersection type combines multiple type requirements into one.
TypeScript uses unions for flexible APIs and control-flow narrowing, and intersections for composing capabilities onto the same value.
Together they let you model "either this or that" and "this and that" directly in the type system.

// TypeScript
type ApiSuccess = {
  ok: true;
  data: { id: string; name: string };
};

type ApiFailure = {
  ok: false;
  error: string;
};

type ApiResponse = ApiSuccess | ApiFailure;

type Timestamped = {
  createdAt: Date;
  updatedAt: Date;
};

type UserRecord = {
  id: string;
  name: string;
} & Timestamped;

function printResponse(response: ApiResponse) {
  if (response.ok) {
    console.log(response.data.name);
  } else {
    console.error(response.error);
  }
}

const user: UserRecord = {
  id: "u_1",
  name: "Allan",
  createdAt: new Date(),
  updatedAt: new Date(),
};
