Observable / reactive stream — Angular

An observable, or reactive stream, is a producer that pushes a sequence of values to its subscribers over time, instead of returning a single value once.
Angular is built on RxJS, where an Observable is composed with operators such as map and filter and only begins emitting when something subscribes to it.
The stream is lazy and push-based: nothing runs until subscription, and each emitted value flows through the operator pipeline down to the subscriber.

// TypeScript (Angular)
import { Component, OnDestroy, OnInit } from '@angular/core';
import { interval, Observable, Subscription } from 'rxjs';
import { filter, map, take } from 'rxjs/operators';

@Component({
  selector: 'app-ticker',
  template: `<p>latest value: {{ latest }}</p>`,
})
export class TickerComponent implements OnInit, OnDestroy {
  // Holds the most recent value pushed by the reactive stream.
  latest = 0;

  // Kept so the stream can be torn down when the component is destroyed.
  private subscription?: Subscription;

  ngOnInit(): void {
    // interval emits 0, 1, 2, ... once per second as a cold (lazy) Observable.
    const ticks$: Observable<number> = interval(1000).pipe(
      filter(n => n % 2 === 0), // keep only even ticks
      map(n => n * 10),         // transform each value as it flows through
      take(5),                  // complete the stream after five emissions
    );

    // Subscribing activates the stream; the callback runs for every value emitted.
    this.subscription = ticks$.subscribe(value => (this.latest = value));
  }

  ngOnDestroy(): void {
    // Unsubscribe to stop the stream and avoid a memory leak after the view is gone.
    this.subscription?.unsubscribe();
  }
}
