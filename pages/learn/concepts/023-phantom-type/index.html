Phantom type

A phantom type is a type parameter or marker that affects compile-time checking even though it has no runtime data of its own.
It is often used to distinguish logically different values that share the same runtime representation.
In TypeScript, this is commonly modeled with branded types that attach a type-only marker to a primitive value.

// TypeScript
type UserId = string & { readonly __brand: "UserId" };
type ProductId = string & { readonly __brand: "ProductId" };

function loadUser(id: UserId) {
  console.log(id);
}

const userId = "u_123" as UserId;
const productId = "p_456" as ProductId;

loadUser(userId);
// loadUser(productId); // type error
