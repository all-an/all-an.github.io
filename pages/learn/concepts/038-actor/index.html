Actor

An actor owns its state and changes it only by processing incoming messages one at a time.
Java often models this with a mailbox queue plus a dedicated worker thread, and libraries such as Akka build richer actor systems on top.
The key idea is message passing instead of direct shared-state mutation.

// Java
import java.util.concurrent.BlockingQueue;
import java.util.concurrent.LinkedBlockingQueue;

sealed interface Msg permits Add, Print {}
record Add(int value) implements Msg {}
record Print() implements Msg {}

BlockingQueue<Msg> mailbox = new LinkedBlockingQueue<>();

Thread actor = Thread.startVirtualThread(() -> {
  int total = 0;
  try {
    while (true) {
      Msg msg = mailbox.take();
      if (msg instanceof Add add) total += add.value();
      if (msg instanceof Print) System.out.println(total);
    }
  } catch (InterruptedException ignored) {
  }
});

try {
  mailbox.put(new Add(2));
  mailbox.put(new Add(3));
  mailbox.put(new Print()); // 5
} catch (InterruptedException ignored) {
}
actor.interrupt();
